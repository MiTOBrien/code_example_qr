Array Iterators

.forEach - iterates over each element in an array letting you do something to (or with) each element. Cannot return a value.


            // EXAMPE #1
            // Given a list of countries, write a function that returns an array of all the country codes using `.forEach`.
            var countries = [
            {
                "countryCode": "AF",
                "countryName": "Afghanistan",
                "population": "29121286",
                "capital": "Kabul",
                "continentName": "Asia"
            },
            {
                "countryCode": "AL",
                "countryName": "Albania",
                "population": "2986952",
                "capital": "Tirana",
                "continentName": "Europe"
            },
            {
                "countryCode": "DZ",
                "countryName": "Algeria",
                "population": "34586184",
                "capital": "Algiers",
                "continentName": "Africa"
            },
            ]
            function getCodes(countries) {
                const countryCodes =[];
                countries.forEach( (country) => {
                    countryCodes.push(country.countryCode);
            })
                return countryCodes;
            }
            console.log(getCodes(countries)) // => ["AF","AL","DZ"]

.filter - iterates over each element in an array and returns a new array of only the elements that match the specified condition


            // EXAMPLE #1
            // Return a new array of the names that are 4 letters or less using the filter prototype method.
            const names = ['nick', 'ben', 'trisha', 'john', 'dustin'];
            // PSEUDOCODE
            // Iterate through array
            // Determine if element is 4 letters or less
            // If yes, Add to new array
            function filterNames() {
                const shortNames = names.filter( (name) => name.length < 5 )
                console.log('result:', shortNames)
            }
            filterNames() 
        
            // Alternate solution
            function FilterNames() {
                const shortNames = names.filter( (name) => {
                    return name.length < 5
                })
                return shortNames
            }
            console.log(filterNames) 
        
            // shortcut/shorthand
            const shortNames = names.filter( (name) => name.length < 5)
            console.log(shortNames)

.map() - iterates over each element in an array and returns a new array of the same length with the modified elements


            // Example #1
            // Using map, iterate over the array of prices, increase the price by $1, and return a new array of increased prices. Store them in a variable called increasedPrices.
            const prices = [4.99, 5.50, 7.00, 10.25]; 
            function increasePrices() {
                const increasedPrices = prices.map( (price) => price + 1 )
                return increasedPrices
            }
            console.log('increased prices:', increasePrices()) 
        
            // Example #2
            // Using an array of temperature data in Fahrenheit, convert the date to degrees Celsius and store them in a variable called degreesC. (T(°C) = (T(°F) - 32) / 1.8)
            const degreesF = [67, 32, 55, 102]
            function convertTemp() {
                const degreesC = degreesF.map( (temperature) => (temperature - 32) / 1.8 )
                return degreesC
            }
            console.log("Temps Celsius:", convertTemp())

.reduce() - iterates over each element in an array and returns a single value (the accumulator) of your specifications


            //Example #1
            // Using reduce, sum up all of the numbers.
            const numbers = [1, 2, 3, 4, 5];
            function sumNumbers() {
            const result = numbers.reduce((acc, num) => {  // If no initial value is set, acc = 1st array element index 0, and num = 2nd array element index 1;
                acc += num;
                return acc;
            }, 0) // optional acc set if you want to set it to something other than the first array element
            return result;
            }
            console.log("Sum of numbers:", sumNumbers())

            // Example #2  *NOTE*  You will see something similar to this in your mid-mod and/or final
            // Using reduce, create a new object the stores the lengths of each word.
            const adjectives = ['fantastic', 'amazing', 'childish'];
            function countChars() {
            const countedChars = adjectives.reduce( (acc, word) => {
                acc[word] = word.length;
                return acc;
            }, {})  // For initial acc - What data type should the final result you're returning be?
            return countedChars;
            }
            console.log("Words with counted characters:", countChars())

            // Example #3
            // Given a list of countries, write a function that returns a new object where the keys are country names and their values are an object containing the rest of the country's data.
            var countries = [
            {
                "countryCode": "AF",
                "countryName": "Afghanistan",
                "population": "29121286",
                "capital": "Kabul",
                "continentName": "Asia"
            },
            {
                "countryCode": "AL",

                "continentName": "Africa"
            },
            ]
            function organizeByCountry(countries) {
                const organizedCountries = countries.reduce((acc, country) => {
                    // Explictly assigns countryName and ...rest is the rest of the object
                    const { countryName, ...rest} = country; 
                    acc[countryName] = rest;
                    return acc;
                }, {})
                return organizedCountries;
            }
            console.log(organizeByCountry(countries))

.sort() - iterates over each element in an array and sorts the elements (in place) based on your specified sorting.


Does a permanent mutation of the original array data; Data is the same, but sorted. 
            // Example 1
            // Sort the numbers in ascending order.
            const sortNumbers = [15, 6, 14, 2, 22, 9, 45];
            function returnSortedNumbers() {
              const sortedNumbers = sortNumbers.sort((a, b) => {
                return a - b;
              })
              return sortedNumbers
            }
            console.log("Sorted ascending:", returnSortedNumbers())
            
            // Example 2
            // Sort the numbers in descending order.
            const sortedDescendingNumbers = [15, 6, 14, 2, 22, 9, 45];
            function returnSortedNumbers() {
              const sortedNumbers = sortNumbers.sort((a, b) => {
                return b - a;
              })
              return sortedNumbers
            }
            console.log("Sorted descending:", returnSortedNumbers())

            // Example #3  *NOTE*  You will see something similar to this in your mid-mod and/or final
            // Write a function sortCitiesByPopulation(cities) that takes an array of cities and uses the sort method to sort the cities based on their population in descending order. Return the array of sorted cities.
            const cities = [
            { name: 'New York', population: 8398748, country: 'United States' },
            { name: 'Tokyo', population: 37393000, country: 'Japan' },
            { name: 'London', population: 8908081, country: 'United Kingdom' },
            { name: 'Mumbai', population: 12478447, country: 'India' }
            ];
            function sortCitiesByPopulation() {
                const sortedCities = cities.sort((city1, city2) => {
                    return city2.population - city1.population;
                    })
                return sortedCities
            }
            console.log(sortCitiesByPopulation("Sorted By Population: ", cities));

Ruby

Testing & Debugging

RSpec:

  • A common testing suite for Ruby.
  • Two folders lib for your program files and spec for your test files.
  • Each library file will have a corresponding _spec.rb test file. dog.rb in your lib directory will have dog_spec.rb in your spec directory
  • rspec spec will run all test files in the spec directory.
  • rspec dog_spec.rb will run the all tests in the file.
  • rspec dog_spec.rb:33 will only run the test that starts on line 33.

pry:

  • Put require 'pry' in your test file or spec_helper.rb file to make it avaialable everywhere
  • Put binding.pry in your code where you want your code to pause so you can examine variable values, etc.

Debugging


2 events will lead to debugging - errors and failures.
  • error: your code won't execute - Typically a typo, undefined variable or method.
  • failure: your code runs/executes but gives an unexpected result. This is typically due to faulty logic in the code you wrote.

Rails

Creating a new rails project: rails new task_manager -T -d="postgresql" --api


Breakdown:
rails new - create a new rails application
task_manager - The folder the project will be created in
-T - We will be using the rails built-in testing suite
-d="postgressql" - We will be using a Postgresql database
--api - We will be building an API instead of an application that has a frontend

Important directories:


app - This is where we configure Models and Controllers
config - Inside this directory, in the routes.rb file is where we will tell our Rails app which HTTP requests to respond to.
db - Where our database structure will be set up.
Gemfile - Tells Rails about any gems we might need to run our app.
Any time the gemfile is updated, you need to tell your application to install or update the additional gems. In the terminal run: bundle install
